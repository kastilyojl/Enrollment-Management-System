// FILE NAMING CONVENTION

// -----------------------
// first letter
// (a)bc_filename

// ADMINISTRATIVE SERVICES

// a > Admission
// e -> Enrollment

// -----------------------
// second letter
// sample: a(b)c_filename

// USER 

// s -> Student
// r -> Registrar
// a -> Accounting
// p -> Professor
// -----------------------

// third letter
// sample: ab(c)_filename

// QUERIES TYPE

// c -> Create
// r -> Read
// u -> Update 
// d -> Delete
// CB -> combination
